Sample Program

math.hxbal

LOGIC
rule add: -> x + y
rule subtract: -> x - y
rule divide: -> x // y
rule multiply: -> x * y
rule modulo: -> x % y

FORCE
    start
        set x = 10
        set y = 5
        ekko(add)
        ekko(subtract)
        ekko(divide)
        ekko(multiply)
        ekko(modulo)
    end

$ cd HexBalance
$ python3 interpreter.py math.hxbal
